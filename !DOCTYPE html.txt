        function showToast(message, type = 'success') {
            const container = document.getElementById('toast-container');
            const toast = document.createElement('div');
            const icon = type === 'success' ? 'fa-check-circle text-green-500' : (type === 'error' ? 'fa-circle-xmark text-red-500' : 'fa-info-circle text-blue-500');
            toast.className = `toast bg-white shadow-xl rounded-2xl border border-gray-100 px-5 py-4 flex items-center gap-3 w-80 mb-2 border-l-4 ${type === 'success' ? 'border-l-green-500' : 'border-l-red-500'}`;

        async function fetchAPI(action, payload = {}) {
            if (IS_PREVIEW) return await mockGASLogic(action, payload);
            const formData = new URLSearchParams(); formData.append('action', action);
            for (let key in payload) formData.append(key, payload[key]);
            try { const response = await fetch(GAS_URL, { method: 'POST', body: formData, headers: { 'Content-Type': 'application/x-www-form-urlencoded' } }); return await response.json(); } catch (error) { return { status: 'error', message: 'Koneksi ke server gagal.' }; }
        }

        async function mockGASLogic(action, payload) {
            await new Promise(r => setTimeout(r, 500));
            if(action === 'login') {
                if(payload.username === 'admin' && payload.password === 'admin2026') return { status:'success', role:'Admin' };
                const user = dummyAkun.find(u => u.username === payload.username && u.password === payload.password);
                if(user) return { status:'success', role: user.role };
                return { status:'error', message:'Login gagal. Cek username/password.' };
            }
            if(action === 'getSettings') return {status:'success', data: globalSettings};
            if(action === 'saveSettings') { globalSettings = JSON.parse(payload.data); return {status:'success'}; }
            if(action === 'getDashboardStats') {
                const td = new Date().toLocaleDateString('en-CA');
                const h = dummyAbsen.filter(a => a.timestamp.startsWith(td) && (a.status==='Hadir')).length;
                return {status:'success', data:{ totalSiswa:dummySiswa.length, hadirHariIni:h, belumHadir:dummySiswa.length-h }};
            }
            if(action === 'getJenjang') return {status:'success', data:dummyJenjang};
            if(action === 'addJenjang') { dummyJenjang.push(JSON.parse(payload.data)); return {status:'success'}; }
            if(action === 'deleteJenjang') { dummyJenjang = dummyJenjang.filter(x=>x.id!==payload.id); return {status:'success'}; }
            
            if(action === 'getKelas') return {status:'success', data:dummyKelas};
            if(action === 'addKelas') { dummyKelas.push(JSON.parse(payload.data)); return {status:'success'}; }
            if(action === 'deleteKelas') { dummyKelas = dummyKelas.filter(x=>x.id!==payload.id); return {status:'success'}; }
            
            if(action === 'getSiswa') return {status:'success', data:dummySiswa};
            if(action === 'addSiswa') { const s = JSON.parse(payload.data); if(dummySiswa.find(x=>x.nis===s.nis)) return {status:'error', message:'NIS ganda'}; dummySiswa.push(s); return {status:'success'}; }
            if(action === 'deleteSiswa') { dummySiswa = dummySiswa.filter(x=>x.nis!==payload.nis); return {status:'success'}; }
            
            if(action === 'getLibur') return {status:'success', data:dummyLibur};
            if(action === 'addLibur') { dummyLibur.push(JSON.parse(payload.data)); return {status:'success'}; }
            if(action === 'deleteLibur') { dummyLibur = dummyLibur.filter(x=>x.tgl!==payload.tgl); return {status:'success'}; }

            if(action === 'getAkun') return {status:'success', data:dummyAkun};
            if(action === 'addAkun') { const a = JSON.parse(payload.data); if(dummyAkun.find(x=>x.username===a.username)) return {status:'error', message:'Username ganda'}; dummyAkun.push(a); return {status:'success'}; }
            if(action === 'deleteAkun') { dummyAkun = dummyAkun.filter(x=>x.username!==payload.username); return {status:'success'}; }

            if(action === 'getRekap') return {status:'success', data:dummyAbsen};
            if(action === 'recordAbsen') {
                const s = dummySiswa.find(x => String(x.nis) === payload.nis.trim()); if(!s) return {status:'error', message:'NIS tidak dikenal'};
                const td = new Date().toLocaleDateString('en-CA');
                const ex = dummyAbsen.find(a=>a.nis===s.nis && a.timestamp.startsWith(td));
                
                if(payload.mode === 'masuk') {
                    if(ex) return {status:'error', message:'Sudah tap Absen Masuk'};
                    dummyAbsen.unshift({timestamp: new Date().toISOString(), nis:s.nis, nama:s.nama, kelas:s.kelas, jenjang:s.jenjang, status:'Hadir', waktuKeluar: ''});
                    return {status:'success', data:s, msg:'Absen Masuk Berhasil'};
                } else {
                    if(!ex) return {status:'error', message:'Belum melakukan Absen Datang!'};
                    if(ex.waktuKeluar) return {status:'error', message:'Sudah tap Absen Pulang'};
                    ex.waktuKeluar = new Date().toISOString();
                    return {status:'success', data:s, msg:'Absen Pulang Berhasil'};
                }
            }
            if(action === 'recordAbsenManual') {
                const req = JSON.parse(payload.data); const exIdx = dummyAbsen.findIndex(a=>a.nis===req.nis && a.timestamp.startsWith(req.tgl));
                const rec = {timestamp: req.tgl+"T00:00:00.000Z", nis:req.nis, nama:req.nama, kelas:req.kelas, jenjang:req.jenjang, status:req.status};
                if(exIdx >= 0) dummyAbsen[exIdx] = rec; else dummyAbsen.unshift(rec); return {status:'success'};
            }
        }

        async function showDashboardDetail(type) {
            document.getElementById('modal-dash-detail').classList.remove('hidden');
            const tbody = document.getElementById('dash-detail-content');
            const title = document.getElementById('dash-detail-title');
            
            tbody.innerHTML = '<tr><td colspan="3" class="p-6 text-center text-gray-400"><i class="fa-solid fa-spinner fa-spin mr-2"></i> Memuat data...</td></tr>';
            
            // Tarik data siswa dan rekap absen jika belum ada di memori
            if(!globalDataSiswa.length) await fetchAPI('getSiswa').then(r => { if(r.status==='success') globalDataSiswa = r.data; });
            if(!globalDataAbsen.length) await fetchAPI('getRekap').then(r => { if(r.status==='success') globalDataAbsen = r.data; });
            
            const td = new Date().toLocaleDateString('en-CA');
            const absenToday = globalDataAbsen.filter(a => String(a.timestamp).startsWith(td));
            
            // Filter hanya siswa yang statusnya Hadir hari ini (bukan Sakit/Izin/Alfa manual)
            const hadirNIS = absenToday.filter(a => a.status === 'Hadir').map(a => String(a.nis).trim());
            
            let filtered = [];
            if(type === 'total') {
                filtered = globalDataSiswa; title.innerText = "Data Keseluruhan Siswa";
            } else if(type === 'hadir') {
                filtered = globalDataSiswa.filter(s => hadirNIS.includes(String(s.nis).trim())); title.innerText = "Daftar Siswa Hadir Hari Ini";
            } else if(type === 'belum') {
                filtered = globalDataSiswa.filter(s => !hadirNIS.includes(String(s.nis).trim())); title.innerText = "Daftar Siswa Belum Hadir / Alfa";
            }
            
            if(filtered.length === 0) {
                tbody.innerHTML = '<tr><td colspan="3" class="p-6 text-center text-gray-400 font-medium">Tidak ada data siswa untuk kategori ini.</td></tr>'; return;
            }
            
            // Urutkan berdasarkan kelas, lalu nama
            filtered.sort((a,b) => a.kelas.localeCompare(b.kelas) || a.nama.localeCompare(b.nama));
            
            tbody.innerHTML = filtered.map(s => `
                <tr class="hover:bg-gray-50 transition">
                    <td class="px-6 py-4 font-mono text-xs text-gray-500">${String(s.nis).trim()}</td>
                    <td class="px-6 py-4 font-bold text-gray-800">${s.nama}</td>
                    <td class="px-6 py-4 text-sm font-bold text-gray-600"><span class="bg-gray-100 text-gray-600 px-3 py-1 rounded-lg border border-gray-200">${s.kelas}</span></td>
                </tr>
            `).join('');
        }

            clockInterval = setInterval(()=>document.getElementById('clock').innerText=new Date().toLocaleTimeString('id-ID'), 1000);
            
            html5QrCode = new Html5Qrcode("reader");
            
            // Konfigurasi ukuran kotak scan yang SUPER RESPONSIF (80% dari lebar layar)
            let qrboxFunction = function(viewfinderWidth, viewfinderHeight) {
                let minEdgePercentage = 0.80; // 80% ukuran layar
                let minEdgeSize = Math.min(viewfinderWidth, viewfinderHeight);
                let qrboxSize = Math.floor(minEdgeSize * minEdgePercentage);
                return {
                    width: qrboxSize,
                    height: qrboxSize
                };
            };

            html5QrCode.start(
                { facingMode: "environment" }, 
                { fps: 15, qrbox: qrboxFunction },
                onScanSuccess, (e) => { /* ignore */ }
            ).catch(() => { showToast("Kamera tidak ditemukan atau diblokir", "error"); closeScanner(); });
        }
        function closeScanner() {
            if(html5QrCode) html5QrCode.stop().then(()=>html5QrCode.clear()).catch(()=>{}); html5QrCode = null;
            clearInterval(clockInterval);
            document.getElementById('view-scan').classList.add('hidden'); document.getElementById('view-main').classList.remove('hidden');
            navigate('dashboard');
        }

        function updateDropdownManualSiswa() {

            }
        }

        function renderTableRekap() {
            const tb = document.getElementById('table-rekap'), th = document.getElementById('thead-rekap'); tb.innerHTML='';
            const mode = document.getElementById('filter-mode').value, tgl = document.getElementById('filter-tanggal').value, bln = document.getElementById('filter-bulan').value, fj = document.getElementById('filter-jenjang-rekap').value, fk = document.getElementById('filter-kelas-rekap').value;
            
            let fSis = globalDataSiswa;
            if(fj!=="") fSis = fSis.filter(s=>s.jenjang===fj); if(fk!=="") fSis = fSis.filter(s=>s.kelas===fk);
            if(!fSis.length) return tb.innerHTML = '<tr><td colspan="6" class="p-6 text-center text-gray-400">Tidak ada siswa.</td></tr>';

            const limitPulang = globalSettings.waktuKeluar || "15:00";
            const isToday = tgl === new Date().toLocaleDateString('en-CA');
            const nowTimeStr = new Date().toLocaleTimeString('en-GB', {hour:'2-digit', minute:'2-digit'});
            const shouldBeAlfa = !isToday || (isToday && nowTimeStr > limitPulang);

            if(mode === 'harian') {
                th.innerHTML = '<th class="px-6 py-4 text-xs font-extrabold text-gray-400 uppercase tracking-widest">Waktu</th><th class="px-6 py-4 text-xs font-extrabold text-gray-400 uppercase tracking-widest">NIS</th><th class="px-6 py-4 text-xs font-extrabold text-gray-400 uppercase tracking-widest">Nama</th><th class="px-6 py-4 text-xs font-extrabold text-gray-400 uppercase tracking-widest">Kelas</th><th class="px-6 py-4 text-xs font-extrabold text-gray-400 uppercase tracking-widest text-center">Status</th>';
                const fRek = globalDataAbsen.filter(a => a.timestamp.startsWith(tgl));
                
                fSis.forEach(s => {
                    const rec = fRek.find(a => String(a.nis) === String(s.nis));
                    let w = "-", b = `<span class="bg-gray-100 text-gray-500 px-3 py-1 rounded-lg text-xs font-bold border border-gray-200">Belum Hadir</span>`;
                    if(rec) {
                        try { const dObj = new Date(rec.timestamp); if(!isNaN(dObj)) w = dObj.toLocaleTimeString('id-ID'); } catch(e){}
                        if(rec.status==='Sakit') b = `<span class="bg-purple-100 text-purple-700 px-3 py-1 rounded-lg text-xs font-bold border border-purple-200">Sakit</span>`;
                        else if(rec.status==='Izin') b = `<span class="bg-teal-100 text-teal-700 px-3 py-1 rounded-lg text-xs font-bold border border-teal-200">Izin</span>`;
                        else if(rec.status==='Alfa') b = `<span class="bg-red-100 text-red-700 px-3 py-1 rounded-lg text-xs font-bold border border-red-200">Alfa</span>`;
                        else {
                            if(shouldBeAlfa && (!rec.waktuKeluar || rec.status==='Hadir')) b = `<span class="bg-orange-100 text-orange-700 px-3 py-1 rounded-lg text-xs font-bold border border-orange-200">Pulang Duluan</span>`;
                            else b = `<span class="bg-green-100 text-green-700 px-3 py-1 rounded-lg text-xs font-bold border border-green-200">Hadir</span>`;
                        }
                    } else if(shouldBeAlfa) b = `<span class="bg-red-100 text-red-700 px-3 py-1 rounded-lg text-xs font-bold border border-red-200">Alfa</span>`;
                    
                    tb.innerHTML += `<tr class="hover:bg-gray-50 border-b border-gray-50 transition"><td class="px-6 py-4 text-xs font-semibold text-gray-500">${w}</td><td class="px-6 py-4 font-mono text-xs text-gray-500">${String(s.nis).trim()}</td><td class="px-6 py-4 font-bold text-gray-800">${s.nama}</td><td class="px-6 py-4 text-sm font-semibold text-gray-600">${s.kelas}</td><td class="px-6 py-4 text-center">${b}</td></tr>`;
                });
            } else {
                th.innerHTML = '<th class="px-6 py-4 text-xs font-extrabold text-gray-400 uppercase tracking-widest">Nama</th><th class="px-6 py-4 text-xs font-extrabold text-green-500 uppercase tracking-widest text-center">H</th><th class="px-6 py-4 text-xs font-extrabold text-purple-500 uppercase tracking-widest text-center">S</th><th class="px-6 py-4 text-xs font-extrabold text-teal-500 uppercase tracking-widest text-center">I</th><th class="px-6 py-4 text-xs font-extrabold text-red-500 uppercase tracking-widest text-center">A</th><th class="px-6 py-4 text-xs font-extrabold text-blue-500 uppercase tracking-widest text-center">%</th>';
                const [y,m] = bln.split('-'); const numDays = new Date(y, m, 0).getDate();
                let hariAktif = 0;
                for(let i=1; i<=numDays; i++) {
                    const d = new Date(y, m-1, i); if(d > new Date()) break;
                    if(d.getDay()===0) continue;
                    if(globalDataLibur.some(l=>l.tgl === d.toLocaleDateString('en-CA'))) continue;
                    hariAktif++;
                }
                const mRec = globalDataAbsen.filter(a => a.timestamp.startsWith(bln));
                fSis.forEach(s => {
                    const sRec = mRec.filter(a=>String(a.nis)===String(s.nis));
                    const h = sRec.filter(a=>a.status==='Hadir').length, sk = sRec.filter(a=>a.status==='Sakit').length, iz = sRec.filter(a=>a.status==='Izin').length;
                    const a = Math.max(0, hariAktif - h - sk - iz);
                    const p = hariAktif>0 ? Math.round((h/hariAktif)*100) : 0;
                    tb.innerHTML += `<tr class="hover:bg-gray-50 border-b border-gray-50 transition"><td class="px-6 py-4"><div class="font-bold text-gray-800">${s.nama}</div><div class="text-xs text-gray-400">${s.kelas}</div></td><td class="px-6 py-4 text-center font-bold text-green-600">${h}</td><td class="px-6 py-4 text-center font-bold text-purple-600">${sk}</td><td class="px-6 py-4 text-center font-bold text-teal-600">${iz}</td><td class="px-6 py-4 text-center font-bold text-red-600">${a}</td><td class="px-6 py-4 text-center font-black ${p>=80?'text-blue-600':'text-orange-500'}">${p}%</td></tr>`;
                });
            }
        }
